### Auto-Populate Logic

```typescript
const autoPopulateSocialLinks = (socialAccounts: Record<Platform, SocialAccount | null>) => {
  const socialLinks: SocialBioLink[] = [];
  
  Object.entries(socialAccounts).forEach(([platform, account]) => {
    if (account?.connected && account?.accountUsername) {
      const url = getPlatformUrl(platform as Platform, account.accountUsername);
      socialLinks.push({
        id: `social-${platform}`,
        platform: platform as Platform,
        url,
        username: account.accountUsername,
        isActive: true,
        clicks: 0,
        autoGenerated: true,
      });
    }
  });
  
  return socialLinks;
};

const getPlatformUrl = (platform: Platform, username: string): string => {
  const urlMap: Record<Platform, string> = {
    Instagram: `https://instagram.com/${username}`,
    TikTok: `https://tiktok.com/@${username}`,
    X: `https://x.com/${username}`,
    Threads: `https://threads.net/@${username}`,
    YouTube: `https://youtube.com/@${username}`,
    LinkedIn: `https://linkedin.com/in/${username}`,
    Facebook: `https://facebook.com/${username}`,
    Pinterest: `https://pinterest.com/${username}`,
    Discord: `https://discord.gg/${username}`, // May need custom handling
    Telegram: `https://t.me/${username}`,
    Reddit: `https://reddit.com/user/${username}`,
  };
  
  return urlMap[platform] || `https://${platform.toLowerCase()}.com/${username}`;
};
```

### Follower Count Calculation

```typescript
const calculateTotalFollowers = (socialAccounts: Record<Platform, SocialAccount | null>): number => {
  let total = 0;
  
  Object.values(socialAccounts).forEach(account => {
    if (account?.connected && account?.followers) {
      total += account.followers;
    }
  });
  
  return total;
};

const formatFollowerCount = (count: number): string => {
  if (count >= 1000000) {
    return `${(count / 1000000).toFixed(1)}M`;
  }
  if (count >= 1000) {
    return `${(count / 1000).toFixed(1)}K`;
  }
  return count.toString();
};
```
